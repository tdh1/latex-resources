\newglossaryentry{digitalthread}
{
	name={Digital Thread},
	description={An integrated information flow that connects all of the phases of the product lifecycle using an accepted authoritative data source (e.g., technical data package, \gls{3d} \gls{cad} model)}
}

\newglossaryentry{digitaltwin}{
	name={Digital Twin},
	description={An integrated model, enabled by the Digital Thread, that combined data from both the cyber-space and physical-space to mirror and predict ``things'' (e.g., activities, performance, outcomes, events) over the life of the model's corresponding physical twin.} 
}

\newglossaryentry{lifecycle}{
	name={Product Lifecycle},
	description={A high-level activity model that starts with marketing, continues through design and manufacture, to selling, then support, and ends with the decommission and disposal or recycle of the product.} 
}

\newglossaryentry{PLM}{
	name={Product Lifecycle Management (PLM)},
	description={The business activity of managing, in the most effective way, a company's products all the way across their lifecycles; from the very first idea for a product all the way through until it is retired and disposed of.} 
}

\newglossaryentry{CPS}{
	name={Cyber-Physical System (CPS)},
	description={Engineered systems that are built from, and depend upon, the seamless integration of computational algorithms and physical components.} 
}

\newglossaryentry{stagegate}{
	name={Stage Gate Process},
	description={Used to describe a point in a project or plan at which development can be examined and any important changes or decisions relating to costs, resources, profits, etc. can be made.} 
}

\newglossaryentry{linkeddata}{
	name={Linked Data},
	description={A method of publishing structured data so that it can be interlinked and become more useful through semantic queries.} 
}
